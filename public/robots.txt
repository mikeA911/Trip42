import { NextApiRequest, NextApiResponse } from 'next';

export default async function handler(
  req NextApiRequest,
  res NextApiResponse
) {
  const baseUrl = 'httpstrip42journals.com';
  
   Fetch dynamic journal slugs from database
   const { data journals } = await supabase
     .from('journals')
     .select('slug, updated_at')
     .eq('privacy', 'public')
     .eq('is_published', true);

  const staticPages = [
    { url '', priority '1.0', changefreq 'daily' },
    { url 'about', priority '0.9', changefreq 'weekly' },
    { url 'features', priority '0.8', changefreq 'weekly' },
    { url 'pricing', priority '0.8', changefreq 'weekly' },
    { url 'themes', priority '0.7', changefreq 'weekly' },
    { url 'blog', priority '0.7', changefreq 'daily' },
    { url 'contact', priority '0.6', changefreq 'monthly' },
  ];

   Example dynamic journal pages
  const dynamicPages = [
     journals.map(j = ({
       url `journals${j.slug}`,
       priority '0.7',
       changefreq 'weekly',
       lastmod j.updated_at
     }))
  ];

  const sitemap = `xml version=1.0 encoding=UTF-8
urlset xmlns=httpwww.sitemaps.orgschemassitemap0.9
  ${staticPages.map(page = `
  url
    loc${baseUrl}${page.url}loc
    lastmod${new Date().toISOString()}lastmod
    changefreq${page.changefreq}changefreq
    priority${page.priority}priority
  url
  `).join('')}
  ${dynamicPages.map(page = `
  url
    loc${baseUrl}${page.url}loc
    lastmod${page.lastmod  new Date().toISOString()}lastmod
    changefreq${page.changefreq}changefreq
    priority${page.priority}priority
  url
  `).join('')}
urlset`;

  res.setHeader('Content-Type', 'textxml');
  res.write(sitemap);
  res.end();
}

import React from 'react';
import { Helmet } from 'react-helmet-async';

interface SEOProps {
  title string;
  description string;
  canonical string;
  ogImage string;
  ogType 'website'  'article';
  keywords string;
  author string;
  publishedTime string;
  modifiedTime string;
  noindex boolean;
  schema object;
}

export const SEO React.FCSEOProps = ({
  title,
  description,
  canonical = 'httpstrip42journals.com',
  ogImage = 'httpstrip42journals.comog-image-default.jpg',
  ogType = 'website',
  keywords = 'travel journal, voice recording, AI translation, travel notes, digital diary',
  author = 'Trip42 Journals',
  publishedTime,
  modifiedTime,
  noindex = false,
  schema
}) = {
  const fullTitle = title.includes('Trip42')  title  `${title}  Trip42 Journals`;
  
  return (
    Helmet
      { Primary Meta Tags }
      title{fullTitle}title
      meta name=title content={fullTitle} 
      meta name=description content={description} 
      meta name=keywords content={keywords} 
      link rel=canonical href={canonical} 
      
      { Author }
      meta name=author content={author} 
      
      { Robots }
      {noindex && meta name=robots content=noindex, nofollow }
      {!noindex && meta name=robots content=index, follow }
      
      { Open Graph }
      meta property=ogtype content={ogType} 
      meta property=ogurl content={canonical} 
      meta property=ogtitle content={fullTitle} 
      meta property=ogdescription content={description} 
      meta property=ogimage content={ogImage} 
      meta property=ogsite_name content=Trip42 Journals 
      
      { Article specific }
      {ogType === 'article' && publishedTime && (
        meta property=articlepublished_time content={publishedTime} 
      )}
      {ogType === 'article' && modifiedTime && (
        meta property=articlemodified_time content={modifiedTime} 
      )}
      
      { Twitter }
      meta property=twittercard content=summary_large_image 
      meta property=twitterurl content={canonical} 
      meta property=twittertitle content={fullTitle} 
      meta property=twitterdescription content={description} 
      meta property=twitterimage content={ogImage} 
      meta name=twittercreator content=@trip42journals 
      
      { Schema.org }
      {schema && (
        script type=applicationld+json
          {JSON.stringify(schema)}
        script
      )}
    Helmet
  );
};

export function JournalPage({ journal }) {
  const journalSchema = {
    @context httpsschema.org,
    @type Article,
    headline journal.title,
    description journal.subtitle,
    image journal.cover_image_url,
    datePublished journal.published_at,
    dateModified journal.updated_at,
    author {
      @type Person,
      name Trip42 User
    },
    publisher {
      @type Organization,
      name Trip42 Journals,
      logo {
        @type ImageObject,
        url httpstrip42journals.comlogo.png
      }
    }
  };

  return (
    
      SEO
        title={journal.title}
        description={journal.subtitle  `Travel journal ${journal.title}`}
        canonical={`httpstrip42journals.comjournals${journal.slug}`}
        ogImage={journal.cover_image_url}
        ogType=article
        publishedTime={journal.published_at}
        modifiedTime={journal.updated_at}
        schema={journalSchema}
        keywords={`${journal.title}, travel journal, ${journal.entries.map(e = e.tags).flat().join(', ')}`}
      
      
      { Your journal content }
    
  );
}

 Lazy load images
export function OptimizedImage({ src, alt, ...props }) {
  return (
    img
      src={src}
      alt={alt}
      loading=lazy
      decoding=async
      {...props}
    
  );
}

 Preload critical resources
export function preloadCriticalResources() {
  if (typeof window !== 'undefined') {
    const link = document.createElement('link');
    link.rel = 'preload';
    link.as = 'font';
    link.href = 'fontsyour-font.woff2';
    link.crossOrigin = 'anonymous';
    document.head.appendChild(link);
  }
}
